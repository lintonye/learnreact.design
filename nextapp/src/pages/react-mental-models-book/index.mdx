### 第 135 页

```tsx
function App() {
  let [query, setQuery] = useState(query)
  useEffect(() => {
    document.title = `查询条件：${query}`
    fetch(`http://some-api.com/search=${query}`).then(doSomething)
  }, [query])
  return <input value={query} placeholder="查询条件" />
}
```
